public static int FindMinArrowShots(int[][] points) {
    //Step 1: Sort the points based on the start point of the balloon
    Array.Sort(points, (a, b) => a[0].CompareTo(b[0]));

    int[] prevPoint = points[0];
    int arrowCount = 0;

    for (int i = 0; i < points.Length; i++)
    {
        int[] currentPoint = points[i];
        if (prevPoint[1] < currentPoint[0])
        {
            arrowCount++;
            prevPoint = currentPoint;
        }
        else
        {
            prevPoint[0] = Math.Max(prevPoint[0], currentPoint[0]);
            prevPoint[1] = Math.Min(prevPoint[1], currentPoint[1]);
        }
    }
    
    arrowCount++;
    return arrowCount;
}
