public class Solution {
    public IList<string> FullJustify(string[] words, int maxWidth)
    {
        var result = new List<string>();
        var sb = new StringBuilder();
        int i = 0;

        while (i < words.Length)
        {
            // --- grouping ---
            int totalLen = words[i].Length;
            int end = i;

            while (end + 1 < words.Length && totalLen + 1 + words[end + 1].Length <= maxWidth)
            {
                end++;
                totalLen += 1 + words[end].Length;
            }

            // --- formatting ---
            int numberOfWords = end - i + 1;
            int gap = numberOfWords - 1;
            int wordsOnlyLen = totalLen - gap;
            int totalSpaces = maxWidth - wordsOnlyLen;
            bool isLastLine = end == words.Length - 1;

            if (isLastLine || gap == 0)
            {
                // left justified
                for (int j = i; j <= end; j++)
                {
                    sb.Append(words[j]);
                    if (j < end) sb.Append(' ');
                }
                sb.Append(new string(' ', maxWidth - sb.Length));
            }
            else
            {
                // full justification
                int spacesPerGap = totalSpaces / gap;
                int extraSpaces = totalSpaces % gap;

                for (int j = i; j <= end - 1; j++)
                {
                    sb.Append(words[j]);
                    sb.Append(new string(' ', spacesPerGap + (j - i < extraSpaces ? 1 : 0)));
                }
                sb.Append(words[end]);
            }

            result.Add(sb.ToString());
            sb.Clear();
            i = end + 1;
        }

        return result;
    }
}
