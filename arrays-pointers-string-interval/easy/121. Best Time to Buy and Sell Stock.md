public class Solution {
    public int MaxProfit(int[] prices) {
        int maxProfit = 0;
        if(prices.Length < 2) return 0;
        
        int minPrice = int.MaxValue;

        foreach (var price in prices)
        {
            if (price < minPrice)
            {
                minPrice = price;
            }
            else
            {
                maxProfit = Math.Max(maxProfit,price - minPrice);
            }
        }

        return maxProfit;
    }
}
