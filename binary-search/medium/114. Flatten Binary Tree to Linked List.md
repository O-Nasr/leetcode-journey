## Problem Statement

Given the root of a binary tree, flatten the tree into a "linked list":

The "linked list" should use the same TreeNode class where the right child pointer points to the next node in the list and the left child pointer is always null.
The "linked list" should be in the same order as a pre-order traversal of the binary tree.


**Example 1:**
```

Input: root = [1,2,5,3,4,null,6]
Output: [1,null,2,null,3,null,4,null,5,null,6]
```

**Example 2:**
```
Input: root = []
Output: []
```

**Example 3:**
```
Input: root = [0]
Output: [0]
```


**Constraints:**
- The number of nodes in the tree is in the range [0, 2000].
- -100 <= Node.val <= 100

private Queue<TreeNode> queue = new Queue<TreeNode>();
public void Flatten(TreeNode root)
{
    FlatTreeUsingQueue(root);
    
    if(queue.Count > 0) root = queue.Dequeue();
    while(queue.Count > 0)
    {
        TreeNode node = queue.Dequeue();
        root.left = null;
        root.right = node;
        root = node;
    }
}
private void FlatTreeUsingQueue(TreeNode root)
{
    if(root == null) return;
    
    queue.Enqueue(root);
    FlatTreeUsingQueue(root.left);
    FlatTreeUsingQueue(root.right);
}
```

public void Flatten(TreeNode root)
{
    var current = root;
    while (current != null)
    {
        if (current.left != null)
        {
            // Find the rightmost node of the left subtree
            var predecessor = current.left;
            while (predecessor.right != null)
            {
                predecessor = predecessor.right;
            }

            // Rewire connections
            predecessor.right = current.right;
            current.right = current.left;
            current.left = null;
        }
        current = current.right;
    }
}
```
