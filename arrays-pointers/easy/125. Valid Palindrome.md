public static bool IsPalindrome(string s)
{
    StringBuilder sb = new StringBuilder();
    foreach (char c in s)
    {
        if(char.IsLetterOrDigit(c)) sb.Append(char.ToLower(c));
    }

    int i = 0;
    int j = sb.Length - 1;

    while (i < j)
    {
        if (sb[i] != sb[j]) return false;
        
        i++;
        j--;
    }
    
    Console.WriteLine(sb.ToString());
    return true;
}
