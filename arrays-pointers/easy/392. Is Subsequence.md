public static bool IsSubsequence(string s, string t)
{
    int sLength = s.Length;
    if(sLength == 0) return true;
    int j = 0;

    foreach (var t1 in t)
    {
        if(j >= sLength) break;
        if(s[j] == t1) j++;
    }
    
    return (sLength == j);
}
