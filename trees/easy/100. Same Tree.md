## Problem Statement

Given the roots of two binary trees p and q, write a function to check if they are the same or not.

Two binary trees are considered the same if they are structurally identical, and the nodes have the same value.


**Example 1:**
```

Input: p = [1,2,3], q = [1,2,3]
Output: true
```

**Example 2:**
```
Input: p = [1,2], q = [1,null,2]
Output: false
```


**Constraints:**
- The number of nodes in both trees is in the range [0, 100].
- 104 <= Node.val <= 104

public static bool IsSameTree(TreeNode p, TreeNode q)
{
    List<int?> pL = GetTreeInArray(p);
    Console.WriteLine("");
    List<int?> qL = GetTreeInArray(q);
    
    if(pL.Count != qL.Count) return false;

    for (int i = 0; i < pL.Count; i++)
    {
        if(pL[i] != qL[i]) return false;
    }
    
    return true;
}

private static List<int?> GetTreeInArray(TreeNode root) 
{
    var result = new List<int?>();
    Queue<TreeNode?> queue = new Queue<TreeNode?>();
    queue.Enqueue(root);

    while (queue.Count > 0) {
        int size = queue.Count;
        for (int i = 0; i < size; i++) {
            Console.Write(queue.Peek()?.val + ", ");
            result.Add(queue.Peek()?.val);
            TreeNode treeNode = queue.Dequeue();
            if(treeNode == null || treeNode.left == null && treeNode.right == null) continue;
            queue.Enqueue(treeNode.left);
            queue.Enqueue(treeNode.right);
        }
    }
    return result;
}
